How to remove remote tag
/usr/libexec/git-core/git-check-ref-format dev-dotnet/system-web/system-web-4.6.182.tar.gz
cd ${DISTDIR} # /var/calculate/remote/distfiles

cd /var/tmp/aaa
git clone -b master --single-branch https://github.com/ArsenShnurkov/shnurise-tarballs.git shnurise-tarballs.git
git remote set-url --push origin https://0000000000000token0000000000000000000000@github.com/ArsenShnurkov/shnurise-tarballs.git
cd /var/tmp/aaa/shnurise-tarballs.git
git checkout --orphan "branch123"
git rm --cached * .gitignore .gitattributes
rm -rf * .gitignore .gitattributes
cp /var/tmp/tarballs/dotnet/dev-dotnet/system-web/system-web-4.6.0.182.tar.gz .
git add *
git commit -m "dev-dotnet/system-web/system-web-4.6.182.tar.gz"
git tag dev-dotnet/system-web/system-web-4.6.182.tar.gz
git push origin dev-dotnet/system-web/system-web-4.6.182.tar.gz
git checkout master
git branch -D branch123
